# Example
Install:

    npm install graviex
	
Example:

        //Import module
        var  graviex  =  require("./graviex.js"); 
	    //Settings / Config  
        graviex.accessKey  =  "";    
        graviex.secretKey  =  ""
        
        graviex.allMarketsTicker(function(res){
	        if(!res.error){        
	            console.log(res)
	        }else{        
	            console.log(res)
	        }        
        });
